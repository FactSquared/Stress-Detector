 6x   = transpose(x(:));
 7imf = [];

13      s1 = getspline(x1);
14      s2 = -getspline(-x1);
15      x2 = x1-(s1+s2)/2;
16      
17      sd = sum((x1-x2).^2)/sum(x1.^2);
18      x1 = x2;
19   end
20

22    imf = x1;
23   end

36function u = isimf(x)
37
38N  = length(x);
39u1 = sum(x(1:N-1).*x(2:N) < 0);
40u2 = length(findpeaks(x))+length(findpeaks(-x));
41if abs(u1-u2) > 1, u = 0;
42else,              u = 1; end
